## 8. PhaseComposer

```kotlin
class PhaseComposer(
    val beatPlanner: BeatPlanner,
    val actorRegistry: ActorRegistry,
    val validator: Validator,
    val perspectiveFilter: PerspectiveFilter
) {
    suspend fun compose(brief: PhaseBrief): StagedPhase {
        val phase = Phase.create(brief)

        while (!phase.isComplete()) {
            val plan = beatPlanner.nextBeat(phase)
            val actorInput = perspectiveFilter.buildActorInput(plan, phase)
            val output = actorRegistry[plan.speaker].generate(actorInput)
            val review = validator.reviewBeat(plan, output, phase)

            if (review.accepted) {
                phase.addBeat(review.toBeat())
            } else {
                phase.applyRepair(review.repairInstruction)
            }
        }

        return phase.stage()
    }
}
```

## 9. Presentation

```kotlin
suspend fun presentPhase(chatId: ChatId, phase: StagedPhase) {
    for (bubble in phase.plannedBubbles) {
        if (flowController.hasForegroundInterruption(chatId)) {
            transaction.cancelRemaining(phase)
            phaseComposer.recomposeFromInterruption(chatId, phase)
            return
        }

        renderer.showTyping(bubble)
        renderer.showBubble(bubble)
        transaction.commitVisibleGatedPatch(bubble)
    }

    transaction.commitPhaseClosePatches(phase)
    ResponsePhase.open(chatId, phase.ResponsePhaseSpec)
}
```
